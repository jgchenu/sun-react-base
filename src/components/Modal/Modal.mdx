## Default Modal

<Playground>

    const [visible, setVisible] = useState(false);
    return (
      <section>
        <Button
          onClick={(e) => {
            setVisible(true);
          }}
        >

## Title Modal

<Playground>
  {() => {
    const title = <div>这是个标题</div>;
    const content = <div>这是内容</div>;
    const [visible, setVisible] = useState(false);
    return (
      <section>
        <Button
          onClick={(e) => {
            e.stopPropagation();
            setVisible(true);
          }}
        >
          显示Title-Content-Footer
        </Button>
        <Modal
          title={title}
          visible={visible}
          okText="确认"
          cancelText="取消"
          onCancel={() => {
            console.log("点击了取消");
            setVisible(false);
          }}
          onOk={() => {
            console.log("点击了确认");
            setVisible(false);
          }}
        >
          <div>显示的内容</div>
        </Modal>
      </section>
    );
  }}
</Playground>
